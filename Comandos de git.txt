
PASO 1 -- CONFIGURAR GIT CON SU NOMBRE DE USUARIO DE GIT Y CORREO
	
git config --global user.name "[name]"
git config --global user.email "[email address]"
git config --global color.ui auto


PASO 2 -- CLONAR EL REPOSITORIO
git clone https://github.com/EduardoIbarra1/IngSoftware-II.git


PASO 3 -- CAMBIAR DE RAMA SIEMPRE ANTES DE TRABAJAR (IMPORTANTE)
git checkout [nombre de su rama]


PASO 4 -- CUANDO TERMINEN DE TRABAJAR EN SUS AVANCES
git add .
git commit -m "MENSAJE"
git push origin [nombre de su rama]
